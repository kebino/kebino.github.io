C - Dynamic Array

May 6, 2022

C does not have a built-in dynamic array like a vector in cpp or arraylist in java. Let’s try to implement something similar to that.

Dynamic Array#

Our dynamic array must be able to resize automatically and hold any data type.

Here’s the header file.

array.h

#ifndef array_h
#define array_h

typedef struct array array; // array handle

array* array_create();                      // constructor
void array_destroy(array** arr);            // destructor
void* array_at(array* arr, int index);      // get item at given index
void array_append(array* arr, void* data);  // insert data
void* array_remove(array* arr, int index);  // remove data at the specified index
int array_get_count(array* arr);            // get item count

#endif

Our array struct is an opaque type, meaning the user of our array can only interact with it through the functions defined in our header.

Here’s the implementation file.

array.c

#include <stdlib.h>
#include "array.h"

struct array {
    void** data;  // The array
    int count;    // Number of items the array is currently holding
    int capacity; // How many slots the array currently has
};

array* array_create() {
    array* arr = malloc(sizeof(array)); // allocate memory for the handle
    arr->count = 0;
    arr->capacity = 4; // initial capacity
    arr->data = calloc(arr->capacity, sizeof(void*)); // allocate memory for the array

    return arr;
}

void array_destroy(array** arr) {
    free((*arr)->data); // free array
    (*arr)->data = NULL;

    free(*arr); // free handle
    *arr = NULL;
} 

void* array_at(array* arr, int index) {
    return arr->data[index];
}

void array_append(array* arr, void* data) {
    // if no space for new item, double the capacity
    if(arr->count == arr->capacity) {
        arr->capacity = arr->capacity * 2;
        arr->data = realloc(arr->data, sizeof(void*) * arr->capacity);
    }
    // append data
    arr->data[arr->count] = data;
    arr->count++;
}

void* array_remove(array* arr, int index) {
    // store item to be removed
    void* removed = arr->data[index];
    
    // index of last item
    int last = arr->count - 1;

    // shift items if item removed 
    // is not the last item
    if(index != last) {
        for(int i = index; i < last; i++) {
            arr->data[i] = arr->data[i + 1];
        }
    }

    arr->data[last] = NULL;
    arr->count--;

    // return removed item
    return removed;
}

int array_get_count(array* arr) {
    return arr->count;
}

The data field in our struct is an array of void pointers. Any data type can be casted to a void* and with that our array can hold any data type.

Example Program#

Here’s an example program that uses or dynamic array.

main.c

#include <stdio.h>
#include "array.h"

int main(int argc, char** args)
{
    // initialize array
    array* arr = array_create();

    // insert items
    array_append(arr, "Bob");
    array_append(arr, "Pedro");
    array_append(arr, "Gringo");
    array_append(arr, "Juan");
    array_append(arr, "Ryu");
    array_append(arr, "Ken");
    array_append(arr, "Vega");
    array_append(arr, "Bison");

    // print items
    for(int i = 0; i < array_get_count(arr); i++) {
        printf("index %d - %s\n", i, (char*)array_at(arr, i));
    }
    printf("no of items = %d\n\n", array_get_count(arr));

    // remove an item
    char* pedro = array_remove(arr, 1);
    
    // print items after remove
    printf("%s was removed\n", pedro);
    for(int i = 0; i < array_get_count(arr); i++) {
        printf("index %d - %s\n", i, (char*)array_at(arr, i));
    }
    printf("no of items = %d\n", array_get_count(arr));

    // destroy array
    array_destroy(&arr);

    return 0;
}

Compile and run:

gcc main.c array.c -o example
./example # or example.exe on windows

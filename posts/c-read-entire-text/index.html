C How to Read the Entire Content of a Text File

Most resources I found for reading files shows how to read a file line by line then display the output, but there are cases where you want to read the entire content of the file and store its contents in a variable for later use.

#include <stdio.h>
#include <stdlib.h>

char* read_file(const char* path)
{
    // open the file
    FILE* f = fopen(path, "rb");

    // determine the length of the file
    fseek(f, 0, SEEK_END);
    long length = ftell(f);
    fseek(f, 0, SEEK_SET);

    // allocate buffer to store the contents of the file
    // size of the buffer should be the file length + 1
    // because we need to add a null terminator at the end
    char* buffer = malloc((sizeof(char) * length) + 1);

    // read the file to our buffer
    fread(buffer, sizeof(char), length, f);

    // at this point file can be closed
    fclose(f);

    // add null terminator
    buffer[length] = '\0';

    return buffer;
}

int main(int argc, char* args[])
{
    // read the file and store its contents to str variable
    char* str = read_file("path_to_file.txt"); 

    // check if all contents are in str
    printf("%s\n", str);

    free(str);
    return 0;
}
